			import turtle

obj = turtle.Pen()					-создать объект(окно со стрелочкой)
obj.forward(px)						-нарисовать линию в пикселях
obj.left(degree)					-изменить направление в градусах
obj.right(degree)
obj.reset()							-очистить холст, черепашка возвращается в начало
obj.clear()							-очистить холст, черепашка остается на месте
obj.backward(px)					-рисует в обратном направлении
obj.up()							-убирает черепашку с холста(черепашка не рисует)
obj.down()							-возвращает черепашку на холст
