git init					взять директорию под контроль git
git add						добавить измененные файлы
git commit					создать новую версию файлов
git status					статус файлов в git
git branch					показать ветки репозитория
git clone					копировать репозиторий
git config
git config --list
git config --global user.name ""
git config --global user.email ""
git checkout					переход между ветками
git merge					добавить состояние файлов в другую ветку
git log						показывает историю версий файлов
git push origin <repository_name> 		выгрузить на сервер

git remote add <repository_name> <url>		привязать к удаленному репозиторию
git pull <repository_name> <branch>		скачать версию с удаленного репозитория
git clean -i					cleans untracked files (-i means interactive)
git revert
